Order by là từ khóa được sử dụng để sắp xếp các kết quả-bộ tăng dần hoặc giảm dần.
            SELECT * FROM Customers
            ORDER BY Country DESC; 
            câu lệnh tăng dần
            ORDER BY Country  DESC
            câu lệnh giảm dần 

            INSERT INTO table_name (column1, column2, column3, ...)
            VALUES (value1, value2, value3, ...);
                    Cú pháp 2 :
                    INSERT INTO table_name
            VALUES (value1, value2, value3, ...);
                    Ví dụ thêm 1 bản ghi mới vào bảng customers ( cột id thì bỏ qua id sẽ tự sinh )
                    ( cần cột nào thì thêm cột đó )
                    INSERT INTO `customers` (`company`, `last_name`, `first_name`, `job_title`)
                    VALUES ('Test', 'dat', 'do', 'dev')


                    Câu lệnh UPDATE ( cập nhật dữ liệu mới )
                    Cú pháp :
                    UPDATE table_name
                    SET column1 = value1, column2 = value2, ...
                    WHERE condition;
                    Ví dụ :
                    UPDATE customers SET first_name = "test 1", last_name = "test 2" WHERE id = 1
                    Câu lệnh DELETE ( xoá dữ liệu )
                    cú pháp : DELETE FROM table_name WHERE condition;
                    ví dụ : DELETE FROM customers WHERE id = 30
                    ( xoá bản ghi có id = 1 )
                    1 - từ khoá MIN lấy giá trị nhỏ nhất
                    ví dụ :
                    SELECT MIN(id) FROM customers
                    lấy ra id nhỏ nhất trong bảng
            
                    2 - từ khoá MAX lấy giá trị lớn nhất
                    SELECT MAX(id) FROM customers
                    lấy ra id lớn nhất trong bảng
