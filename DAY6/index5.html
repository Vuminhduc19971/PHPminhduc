            dùng để đếm

            total lúc này sẽ trờ thành 1 cột ảo khi truy vấn

            SELECT AVG(list_price) FROM products

            từ khoá IN cho phép bạn lọc nhiều giá trị trong 1 cột với mệnh đề where
            từ khoá IN là viết tắt của mệnh đề where or
            cú pháp :
    
            SELECT column_name(s)
    FROM table_name
    WHERE column_name IN (value1, value2, ...);
    
            ví dụ :
            SELECT * FROM customers WHERE id IN(2,3,4)
            lất ra các bản ghi có id là 2 hay 3 hay 4
       
             Từ khóa Between (Lấy giá trị trong khoảng)
            SELECT colum name(s)
            FROM  table_name
            WHERE column_name BETWEEN value1 AND value2;
            ví dụ:
            SELET * FROM customers WHERE id BETWEEN 1 AND 10
            lấy ra các bản ghi trong bảng khách hàng có id  từ 1 đén 10
            
            Từ khoá GROUP BY được sử dụng để gom nhóm khi có 1 điểm chung vào 1 dòng tóm tắt
        ví dụ như lấy ra tất cả các khách hàng đến từ mỗi quốc gia
        Từ khoá GROUP BY thường được dùng cùng các hàm trong SQL như (COUNT, MAX, MIN, SUM, AVG)
        ví dụ :
        tính tổng số khách hàng đến từ các thành phố
        SELECT city , COUNT(id) AS total FROM customers GROUP BY city
